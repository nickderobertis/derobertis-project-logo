[docs]def build_logos(logos: Sequence[ProjectLogo] = LOGOS, out_folder: str = '.'):
    for logo in logos:

[docs]def build_rst(logos: Sequence[ProjectLogo] = LOGOS, out_folder: str = '.', images_folder: str = '.'):
    out_path = os.path.join(out_folder, 'creations.rst')
    rst = "Created Logos\n*****************"
    for logo in logos:
        rst += '\n'
        rst += logo.rst(images_folder)
    with open(out_path, 'w') as f:
        f.write(rst)
